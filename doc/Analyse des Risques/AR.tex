\section{Evaluation des risques potentiels}

\subsection{Classification des points durs selon leur importance}
Intitulé [\emph{membre(s) concerné(s)}] :
\begin{enumerate}
	\item Etude des RFC décrivant les systèmes / Etat de l'art.
	\item Démonstration de la sécurité des systèmes implémentés [\emph{Resp. qualité}].
	\item Planifier l'évolution du projet [\emph{Chef de projet}].
	\item Communication régulière avec le client [\emph{MOA}].
	\item Formation / suivi des équipes [\emph{Resp. technique / Resp. qualité / Chef de projet}].
	\item Surveiller l'évolution des spécifications [\emph{MOA}].
	\item Emploi de technologies spécifiques (Android, Java Card).
	\item Communication au sein de l'équipe [\emph{Chef de projet}].
	\item Mutualisation des connaissances et des compétences.
	\item Gestion globale du projet (documentation).
	\item Veuiller au besoins techniques des membres de l'équipe [\emph{Chef de projet}].
\end{enumerate}

\subsection{Points durs \& risques associés}

\begin{enumerate}
	\item
		\begin{description}
			\item [Point dur :] Etude des RFC décrivant les systèmes / Etat de l'art.
			\item [Risques associés :] --
				\begin{itemize}
					\item Le développement effectué ne correspond pas aux RFC.
					\item Etat de l'art pas assez exhaustif.

		\begin{description}

		\end{description}
	\item
		\begin{description}
			\item [Point dur :] Planifier l'évolution du développement du projet.
			\item [Risque associé :] Impondérables (surcharge de travail).

	\item
		\begin{description}

			\item [Risques associés :] ---
				\begin{itemize}

					\item Passer à côté d'une demande du client.
					\item Les clients font faillite.
				\end{itemize}
		\end{description}
	\item
		\begin{description}
			\item [Point dur :] Formation / suivi des équipes [\emph{Resp. technique / Resp. qualité / Chef de projet}].
			\item [Risque associé :] Retard d'une partie de l'équipe sur la compréhension théorique / les compétences techniques.
		\end{description}
	\item
		\begin{description}
			\item [Point dur :] Surveiller l'évolution des spécifications [\emph{MOA}].
			\item [Risques associés :] --
				\begin{itemize}
					\item Spécification incomplète.
					\item Spécification trop exigentes (Problème pour atteindre la performance désirée).

	\item
		\begin{description}
			\item [Point dur :] Emploi de technologies spécifiques (Android, Java Card).
			\item [Risque associé :] Appréhension des nouvelles technologies plus longue que prévue.
		\end{description}
	\item
		\begin{description}
			\item[Point dur :] Communication au sein de l'équipe [\emph{Chef de projet}].
			\item[Risques associés :] --
				\begin{itemize}
					\item Chef de projet incompétent.
					\item Conflit au sein de l'équipe.

		\begin{description}

			\item [Risque associé :] Départ d'une ou plusieurs personnes clé du projet.
		\end{description}

		\begin{description}
			\item [Point dur :] Gestion globale du projet (documentation).

					\item Phase de développement plus longue que prévu.
					\item Le développement effectué ne correspond pas à la demande du client.
					\item Sous estimation des efforts nécessaires à la réalisation du projet.
				\end{itemize}
		\end{description}
	\item
		\begin{description}
			\item [Point dur :] Veuiller au besoins techniques des membres de l'équipe [\emph{Chef de projet}].
			\item [Risques associés :] --

					\item Un membre se retrouve privé de ses outils de travail.
					\item Problème avec le git.
				\end{itemize}
		\end{description}
\end{enumerate}

\section{Gestion des risques les plus importants}

\subsection{Top 5}
\begin{itemize}
	\item Risque 5
		\begin{description}
			\item [Evènement/situation redouté :] Retard d'une partie de l'équipe sur la compréhension théorique / les compétences techniques.
			\item [Point dur associé :] Formation / suivi des équipes [\emph{Resp. technique / Resp. qualité / Chef de projet}].
			\item [Impact :] Préoccupant.
			\item [Probabilité :] Forte.
			\item [Plan de réduction de risque :] Suivi régulier des équipes par le responsable technique et le chef de projet.
			\item [Plan d'action :] Consultation d'un responsable technique. Si aucune amélioration, re-manier l'équipe concernée (ajout d'une personne ayant suffisamment avancé dans son travail, le remplacement ne semblant pas être une bonne solution).
		\end{description}
	\item Risque 4
		\begin{description}
			\item [Evènement/situation redouté :] Appréhension des nouvelles technologies plus longue que prévue.
			\item [Point dur associé :] Emploi de technologies spécifiques (Android, Java Card).
			\item [Impact :] Préoccupant.
			\item [Probabilité :] Forte.
			\item [Plan de réduction de risque :] Suivi régulier des équipes par le responsable technique et le chef de projet. Rédaction de rapport explicatifs quant à la technologie étudiée (généralités, fonctionnalités nécéssaires et/ou utiles pour le projet, particularités...).
			\item [Plan d'action :] Ajout d'une personne à l'équipe concernée pour combler les lacunes, un remplacement est envisageable.
		\end{description}
	\item Risque 3
		\begin{description}
			\item [Evènement/situation redouté :] Faille de sécurité découverte après test.
			\item [Point dur associé :] Démonstration de la sécurité des systèmes implémentés [\emph{Resp. qualité}].
			\item [Impact :] Important.
			\item [Probabilité :] Moyenne.
			\item [Plan de réduction de risque :] Suivi ($\Rightarrow$ tests) régulier des fonctionnalités du produit.
			\item [Plan d'action :] Renvoie d'un rapport d'erreur détaillé à l'équipe ayant développé le sujet du test pour que celle-ci puisse la corriger dans les plus brefs délais. Si le temps manque, focaliser autant d'équipiers que nécessaire sur le problème.
		\end{description}
	\item Risque 2
		\begin{description}
			\item [Evènement/situation redouté :] Phase de développement plus longue que prévu.
			\item [Point dur associé :] Gestion globale du projet (documentation).
			\item [Impact :] Important.
			\item [Probabilité :] Forte.
			\item [Plan de réduction de risque :] Suivi régulier des équipes par le responsable technique et le chef de projet.
			\item [Plan d'action :] Réunion d'équipe pour analyser et adapter les démarches en vue d'accélérer la productivité.
		\end{description}
	\item Risque 1
		\begin{description}
			\item [Evènement/situation redouté :] Impondérables (surcharge de travail).
			\item [Point dur associé :] Planifier l'évolution du dévelopement du projet [\emph{Chef de projet}].
			\item [Impact :] Important.
			\item [Probabilité :] Forte.
			\item [Plan de réduction de risque :] Raccourcir les délais afin de garder un peu d'avance.
			\item [Plan d'action :] Préparation et ingurgitation massive de caféine pour des nuits blanches de travail. Mise en communs des ressources et des compétences de toutes l'équipe. Prolonger les délais n'est pas envisageable.
		\end{description}
\end{itemize}
